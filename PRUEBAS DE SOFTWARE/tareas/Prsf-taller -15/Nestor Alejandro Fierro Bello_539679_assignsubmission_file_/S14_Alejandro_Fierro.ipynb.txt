        "def seleccionar_tests_optimos(tests, presupuesto):\n",
        "    \"\"\"\n",
        "    Knapsack 0/1 para selección óptima de tests.\n",
        "    tests: [(nombre, tiempo, valor), ...]\n",
        "    Retorna (valor_total, lista_tests_seleccionados, tiempo_usado)\n",
        "    \"\"\"\n",
        "    n = len(tests)\n",
        "    # Tabla DP: dp[i][w] = max valor usando i primeros tests con w minutos\n",
        "    dp = [[0] * (presupuesto + 1) for _ in range(n + 1)]\n",
        "\n",
        "    for i in range(1, n + 1):\n",
        "        nombre, tiempo, valor = tests[i - 1]\n",
        "        for w in range(presupuesto + 1):\n",
        "            dp[i][w] = dp[i - 1][w]          # no tomar el test i\n",
        "            if tiempo <= w:                    # tomar el test i (si cabe)\n",
        "                dp[i][w] = max(dp[i][w], dp[i - 1][w - tiempo] + valor)\n",
        "\n",
        "    # Reconstrucción: ¿qué tests se seleccionaron?\n",
        "    seleccionados = []\n",
        "    w = presupuesto\n",
        "    for i in range(n, 0, -1):\n",
        "        if dp[i][w] != dp[i - 1][w]:\n",
        "            nombre, tiempo, valor = tests[i - 1]\n",
        "            seleccionados.append(nombre)\n",
        "            w -= tiempo\n",
        "    seleccionados.reverse()\n",
        "\n",
        "    tiempo_usado = sum(t[1] for t in tests if t[0] in seleccionados)\n",
        "    return dp[n][presupuesto], seleccionados, tiempo_usado\n",
        "\n",
        "\n",
        "def seleccionar_tests_greedy(tests, presupuesto):\n",
        "    \"\"\"\n",
        "    Greedy: prioriza por ratio valor/tiempo (mayor primero).\n",
        "    \"\"\"\n",
        "    ordenados = sorted(tests, key=lambda t: t[2] / t[1], reverse=True)\n",
        "    seleccionados = []\n",
        "    tiempo_restante = presupuesto\n",
        "    valor_total = 0\n",
        "\n",
        "    for nombre, tiempo, valor in ordenados:\n",
        "        if tiempo <= tiempo_restante:\n",
        "            seleccionados.append(nombre)\n",
        "            tiempo_restante -= tiempo\n",
        "            valor_total += valor\n",
        "\n",
        "    tiempo_usado = presupuesto - tiempo_restante\n",
        "    return valor_total, seleccionados, tiempo_usado\n",
        "\n",
        "print('✅ Funciones DP y Greedy cargadas')"
